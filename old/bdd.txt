CREATE TABLE `registro`.`usuarios` ( `id_usuario` SERIAL NOT NULL , `nombre` VARCHAR(30) NOT NULL , `apellido1` VARCHAR(50) NOT NULL , `apellido2` VARCHAR(50) NOT NULL , `direcion` VARCHAR(100) NOT NULL , `correo_electronico` VARCHAR(40) NOT NULL , `telefono` INT(9) NOT NULL , `fecha_de_nacimiento` DATE NOT NULL , `NIF` VARCHAR(10) NOT NULL , `poblacion` VARCHAR(30) NOT NULL , `provincia` VARCHAR(20) NOT NULL , `password` VARCHAR(10) NOT NULL ) ENGINE = InnoDB;
